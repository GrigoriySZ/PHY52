class Departemnt:
    def __init__(self, name: str):
        self.__name = name
    @property
    def Name(self):
        return self.__name
    @Name.setter
    def Name(self, new_name: str):
        self.__name = new_name

class Employer:
    def __init__(self, fio: str, age: int, salary: int, depart: Departemnt):
        self.__fio = fio
        self.__age = age
        self.__salary = salary
        self.__depart = depart
    def get_instace(f: str, i: str, o: str, age: int, salary: int,
                 depart: Departemnt):
        return Employer(f+i+o, age, salary, depart)

    #магический метод строкового представления объекта
    def __str__(self):
        return f"""ФИО {self.__fio}, 
        Возраст {self.__age}, 
        Зарплата {self.__salary}
        Отдел {self.__depart.Name}"""
    @property
    def Department(self):
        return self.__depart
    @Department.setter
    def Department(self, new_depart: Departemnt):
        self.__depart = new_depart

    def __add__(self, other):
        if isinstance(other, Employer):
            res = Employer(self.__fio,
                        (self.__age + other.__age)// 2,
                         self.__salary + other.__salary,
                          other.__depart )
            return res
